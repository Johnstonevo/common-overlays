Works with game boy / gb color / game gear.  It's a small viewport for those with large display devices.

retroarch.cfg settings for ideal use of the overlay 


# gb, gg, gbc 160x144
custom_viewport_width = 640
custom_viewport_height = 576
custom_viewport_x = 640
custom_viewport_y = 252
video_message_pos_x = 0.34
video_message_pos_y = 0.24
video_fullscreen_x = 1920
video_fullscreen_y = 1080
aspect_ratio_index = 22
input_overlay = "/opt/retropie/emulators/retroarch/overlays/gb.cfg"
input_overlay_enable = true
input_overlay_opacity = 1.000000
input_overlay_scale = 1.000000
video_smooth = "false"
video_scale_integer = "false"
video_shader_enable = "false"
video_font_size = "24"
